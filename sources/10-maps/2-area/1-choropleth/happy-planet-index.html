Ingest

# Data source must have an ISO3 field
df <- read.xlsx("archetypes/happy-planet-index/happy-planet-index-2020.xlsx", "1. All countries", rowIndex = 9:161)
df

Wrangle

select columns and remove NAs

# Only select columns :Country, ISO and HPI for the map
df_wrangle <-df[,c(2,3,10)]
df_wrangle$HPI<-as.numeric(df$HPI)
na.omit(df_wrangle)

World map

from natural earth

# natural earth base map
ne_world <- ne_countries(scale = "small", returnclass = "sf")
# removes Antarctica
ne_world <- filter(ne_world, iso_a3 != "ATA")

Wrangle

merge data and map geometry

# combine data with map based on ISO3 codes
ne_world_df <- merge(ne_world, df_wrangle, by.x = "iso_a3", by.y = "ISO", all = TRUE)

v1 <- ggplot(data = ne_world_df) +
  # map feature styling
  geom_sf(aes(fill=HPI), color="#FFFFFF", stroke=0.5) +
  scale_fill_gradientn(colours = rev(pal)) + 
  coord_sf() +
  theme_bw() +
  labs(x="", # Longitude
       y="", # Latitude
       title = "Choropleth Map", 
       subtitle="The Happy Planet Index 2020") +
  theme_opts

girafe(ggobj = v1, width_svg = 16, height_svg = 9,
       options = list(opts_sizing(rescale = TRUE, width = 1.0))
)

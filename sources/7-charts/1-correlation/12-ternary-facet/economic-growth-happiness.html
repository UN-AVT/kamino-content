Economic growth does not guarantee rising happiness

Ternary Facet

Load the data [Ingest]

# Load data
df <- read.csv("archetypes/economic-growth-happiness/economic-growth-does-not-guarantee-rising-happiness.csv", header = TRUE, stringsAsFactors = FALSE)
df

Prepare the data [Wrangle]

df_wrangle <- df

# Rescale 0 to 100
df_wrangle$happy <- rescale(df_wrangle$happy, to=c(0,100))
df_wrangle$gdp.pc <- rescale(df_wrangle$gdp.pc, to=c(0,100))
df_wrangle$pop <- rescale(df_wrangle$pop, to=c(0,100))

df_wrangle

Plot the chart

USDA National Nutrient Database

continent_palette <- c("Africa" = "#77dff7", "Americas" = "#b5ea32", "Asia" = "#ff5973", "Europe" = "#ffe800", "Oceania" = "#ff5973")

theme_opts <- theme(
  legend.position = "bottom",
  legend.title = element_blank(),
  tern.axis.title.L = element_text(hjust = 0.5),
  tern.axis.title.R = element_text(hjust = 0.5),
  panel.background = element_blank(),
  panel.border = element_blank(),
  plot.background = element_blank(),
  plot.margin = unit(c(0, 0, 0, 0), "in")
)

v1 <- ggtern(data=df_wrangle, aes(gdp.pc, happy, pop)) + 
  geom_point(aes(color = quadrant), size = 2.5, alpha=.6) + 
  #scale_color_manual(values = continent_palette) +
  coord_tern() +
  theme_minimal() +
  theme_nomask() +
  theme_opts +
  labs(title = "Gapminder", 
           subtitle = "2020") +
           xlab("gdp") + 
           ylab("happy") +
           zlab("population") #+
  #facet_wrap(~ paradox, ncol = 1)

girafe(ggobj = v1, width_svg = 12, height_svg = 16,
       options = list(opts_sizing(rescale = TRUE, width = 1.0)))

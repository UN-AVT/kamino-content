Internet Penetration by Region

Marimekko

Load data

df = read.csv("archetypes/internet-penetration-by-region/internet-penetration-by-region.csv", header = TRUE)
df

Prepare data

df$xmax <- cumsum(df$Population_Percent_of_World)
df$xmin <- df$xmax - df$Population_Percent_of_World
df$xlabel <- (df$xmax + df$xmin)/2
df$label <- paste(as.character(df$Penetration_Rate*100),"%")
data<-df %>% rename(Regions=World_Regions)
data

Plot

theme_opts <- theme(
  legend.position = "none",
  legend.title = element_blank(),
  axis.text = element_blank(),
  axis.line = element_blank(),
  axis.ticks = element_blank(),
  axis.title = element_blank(),
  panel.grid = element_blank(),
  panel.grid.major = element_blank(),
  panel.grid.minor = element_blank(),
  panel.background=element_rect(fill="white", colour="white"),
  panel.border = element_blank(),
  plot.background = element_blank(),
  plot.margin = unit(c(0, 0, 1, 0), "in")
)

v1 <-ggplot(data) +
    geom_rect(aes(ymin = 0, ymax = Penetration_Rate, xmin = xmin, xmax = xmax, fill = Regions), color = "white", alpha = 0.5) + 
    geom_text(aes(x = xlabel, y = Penetration_Rate, label = label, vjust = -1), position = "identity", color = "black", size = 3 ) +
    geom_text(aes(x = xlabel, y = 0, label = Regions), hjust = "right", vjust = 0.5, position = "identity", color = "black", size = 3.5, angle = 45 ) +
    geom_hline(yintercept = 0.6) +
    coord_cartesian(clip="off") +
    theme(plot.title = element_text(size = rel(2)), panel.background = element_rect(fill = "white", colour = "grey50")) +
    theme_opts

girafe(ggobj = v1, width_svg = 16, height_svg = 9,
       options = list(opts_sizing(rescale = TRUE, width = 1.0)))

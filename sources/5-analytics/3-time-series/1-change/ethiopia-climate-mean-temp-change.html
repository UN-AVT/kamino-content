Case: Climate Change in Ethiopia

Load the annual mean temperature data

df<-read_csv('archetypes/tas_timeseries_annual_cru_1901-2020_ETH.csv')
df

Multiple Changes in Mean and Variance

df1.pelt=cpt.meanvar(df$Ethiopia,
      method='BinSeg',Q=20,penalty="SIC")
points <- cpts(df1.pelt)

for(i in points){
  print(paste0('Change Point Year :' ,df$Year[i]))
}
## [1] "Change Point Year :1975"
## [1] "Change Point Year :2000"
param.est(df1.pelt)
## $mean
## [1] 22.34507 22.65000 23.20550
## 
## $variance
## [1] 0.07484100 0.03997600 0.04963475
plot(df1.pelt,cpt.width=1,cpt.col='red')

for(i in points){
  abline(v=i, col="blue")
}

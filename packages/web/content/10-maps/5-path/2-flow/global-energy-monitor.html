# tooltip -----
fun_tooltip <- function(dat){
  
  cols <- as.data.frame(colnames(dat))
  colnames(cols) <- "col"
  cols <- filter(cols, col != "point_id")
  
  for (row in 1:nrow(dat)) {

    tooltip <- NULL
    
    for (c in 1:nrow(cols)) {
      tooltip <- paste0(tooltip, cols[c,"col"], ": ", dat[row, cols[c,"col"]], "\n")
    }
    dat[row, "tooltip"] <- tooltip
  }
  return(dat)
}

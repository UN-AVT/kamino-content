Wrangle

filter origin and year

# change the filter to experiment with other exporters or importers
df_flows_origin <- filter( df_flows, Country_of_origin_ISO3c == "SYR" )
df_flows_origin <- filter( df_flows_origin, Refugees_under_UNHCR_mandate > 0 )
df_flows_origin <- filter( df_flows_origin, Year == 2020 )

df_flows_origin <- df_flows_origin %>% select( Country_of_origin, Country_of_origin_ISO3c, Country_of_asylum, Country_of_asylum_ISO3c, Refugees_under_UNHCR_mandate )

head(df_flows_origin, n=10)

Wrangle

enrich with coordinates

centroids = read.csv("archetypes/refugee-flows/country-centroids.csv", header = TRUE, stringsAsFactors = FALSE, as.is=T)
countryref <- centroids %>% select(iso3c, latitude, longitude)
# countryref

df_geo <- merge(df_flows_origin, countryref, by.x="Country_of_origin_ISO3c", by.y="iso3c")
df_geo <- df_geo %>% rename(slongitude = longitude, slatitude = latitude)

df_geo <- merge(df_geo, countryref, by.x="Country_of_asylum_ISO3c", by.y="iso3c")
df_geo <- df_geo %>% rename(tlongitude = longitude, tlatitude = latitude)

head(df_geo, n=10)

v1 <- ggplot(data = ne_world) +
  geom_sf(fill="#ffffff", color="#cccccc", size=0.75) +
  geom_point( data = df_geo, aes(x=tlongitude, y=tlatitude, size=Refugees_under_UNHCR_mandate, fill=Refugees_under_UNHCR_mandate), shape=21, color="white", stroke=1, alpha=1.0) +
  geom_segment( data = df_geo, aes( x=slongitude, y=slatitude, 
                                  xend=tlongitude, yend=tlatitude, 
                                  size=Refugees_under_UNHCR_mandate, color=Refugees_under_UNHCR_mandate, alpha=Refugees_under_UNHCR_mandate ), 
                                  lineend="round", linejoin="round" ) +
  geom_point( data = head(df_geo, n=1), aes(x=slongitude, y=slatitude), shape=21, size=14, fill='black', color="black", stroke=1, alpha=0.7) +
  geom_text( data = head(df_geo, n=1), aes(x=slongitude, y=slatitude, label = Country_of_origin_ISO3c), color="white", alpha=1.0, family = "inconsolata") +
  scale_size_continuous(range=c(1,10), guide="none") +
  scale_alpha_continuous(range=c(0.2,0.6), guide="none") +
  scale_fill_gradient(low = "#3949AB", high = "#8E24AA", guide="none") + 
  scale_color_gradient(low = "#3949AB", high = "#8E24AA", guide="none") + 
  theme_bw() +
  labs(x="",
       y="",
       title = "Refugee Flows", 
       subtitle="From Syria, Refugees_under_UNHCR_mandate") +
  theme_opts

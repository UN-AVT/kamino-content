Ingest the data

Load GDP Dataset

#Load
url_root <- "https://raw.githubusercontent.com/UN-AVT/kamino-source/main/sources/0-shared/data/"

url_file1 <- "road-deaths-by-country/gdp.csv"
url1 <- paste0(url_root, url_file1)

df_gdp <- read.csv(url1, header = TRUE, stringsAsFactors = FALSE, encoding = "UTF-8")
df_gdp

Wrangle

Replace value “NA” with “Not Applicable” using is.na()

#Replacing one value with another 
df_gdp[is.na(df_gdp)] = "Not Applicable"
df_gdp

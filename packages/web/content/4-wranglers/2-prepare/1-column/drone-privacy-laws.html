#Load data
url_root <- "https://raw.githubusercontent.com/UN-AVT/kamino-source/main/sources/0-shared/data/"
url_file <- "drone-privacy-laws/drone-privacy-laws.csv"
url <- paste0(url_root, url_file)

df <- read.csv(url, header = TRUE, stringsAsFactors = FALSE)
df

Wrangle

Combine columns country and continent using unite() function from tidyr package

#renaming first column
names(df)[1] <- "Country"

#combining columns
df1 <- df %>% unite('country_region',Country,Continent, sep = "_", remove = TRUE, na.rm = FALSE )
df1

Combine using paste() function from base R

#combining columns
df2 <- df
df2$country_region <- paste(df$Country, df$Continent, sep="_")
df2
